political_actions = {
	national_unrest_civil_war_imminent = {

		priority = 100

		icon = generic_ignite_civil_war

		available = {
			hidden_trigger = { always = no }
		}

		days_mission_timeout = 70

		fire_only_once = no

		activation = {
			has_civil_war = no
			check_variable = { ebf_economy_national_unrest > 0.8 }
			has_country_flag = ebf_economy_national_unrest_uprising_imminent 
		}

		visible = {
			has_country_flag = ebf_economy_national_unrest_uprising_imminent 
			check_variable = { ebf_economy_national_unrest > 0.8 }
		}

		cancel_if_not_visible = yes

		is_good = no

		timeout_effect = {
			clr_country_flag = ebf_economy_national_unrest_uprising_imminent
			if = {
				limit = {
					has_government = fascism
				}
				start_civil_war = {
					ideology = communism
					size = 0.5
				}
			}
			else = {
				start_civil_war = {
					ideology = fascism
					size = 0.5
				}
			}
			add_stability = -0.15	
			set_variable_to_random = {
				var = ebf_economy_national_unrest
				max = 0.35
				min = 0.15
			}
			
		}
	}
}
