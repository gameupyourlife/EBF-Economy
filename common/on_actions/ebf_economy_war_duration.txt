on_actions = {
	on_startup = {
		effect = {
			every_country = {
				set_variable = { ebf_economy_war_duration = 0 }
			}
		}
	}
	on_monthly = {
		effect = {
			every_country = {
				if = {
					limit = {
						has_war	= yes
					}
					add_to_variable = { ebf_economy_war_duration = 1 }
				}
				else = {
					set_variable = { ebf_economy_war_duration = 0 }
				}
			}
		}
	}
}
