# Healthcare Expenses
country_event = {
	id = ebf_economy_expenses_event.1
	title = ebf_economy_expenses_event.1.t
	desc = ebf_economy_expenses_event.1.d
	picture = GFX_report_event_soviet_soldiers
	
	is_triggered_only = yes
	hidden = yes

    immediate = {
		set_variable = { ebf_economy_healtcare_expenses = ebf_economy_healtcare_rate }
		multiply_variable = { ebf_economy_healtcare_expenses = ebf_economy_population }
		
		set_temp_variable = { ebf_economy_healtcare_expenses_temp = ebf_economy_soldiers_healtcare_rate }
		multiply_temp_variable = { ebf_economy_healtcare_expenses_temp = manpower }
	
		add_to_variable = {
			ebf_economy_healtcare_expenses = ebf_economy_healtcare_expenses_temp
		}
		multiply_variable = {
			ebf_economy_healtcare_expenses = ebf_economy_inflation
		}
	}
}

		}
		set_temp_variable = { ebf_economy_soldiers = manpower_k }
